Color.black.opacity(0.03) : backgroundColor.opacity(0.3), radius: 10, x: 0, y: 5) } } // MARK: - 子視圖: 最近答對紀錄 struct RecentMasteryView: View { // 假數據紀錄 let items = [ ("あ", "ア"), ("か", "カ"), ("が", "ガ"), ("し", "シ") ] var body: some View { VStack(alignment: .leading, spacing: 12) { Text("🎯 最近熟記的配對") .font(.headline) .padding(.leading, 5) HStack(spacing: 10) { ForEach(items, id: \.0) { item in HStack(spacing: 4) { Text(item.0) .fontWeight(.bold) .foregroundColor(.blue) Text("➔") .font(.caption) .foregroundColor(.gray) Text(item.1) .fontWeight(.bold) } .font(.subheadline) .padding(.horizontal, 12) .padding(.vertical, 8) .background(Color.white) .cornerRadius(10) .shadow(color: Color.black.opacity(0.02), radius: 5) } } } .padding(.top, 10) } } // MARK: - 預留的空白目標頁面 (為了讓 Navigation 可運作) // 1. 測驗頁面 struct QuizView: View { var body: some View { VStack { Text("測驗頁面") .font(.largeTitle) Text("(這裡是將來放一分鐘挑戰程式碼的地方)") .foregroundColor(.gray) } .navigationTitle("1 分鐘挑戰") // 設定導覽列標題顯示模式為靠左大標題 .navigationBarTitleDisplayMode(.large) } } // 2. 五十音表頁面 struct KanaTableView: View { var body: some View { VStack { Text("五十音對照表") .font(.largeTitle) Text("(這裡是將來放完整對照表表格的地方)") .foregroundColor(.gray) } .navigationTitle("學習與複習") .navigationBarTitleDisplayMode(.large) } } // MARK: - Xcode 預覽區 (這段在正式 App 不會執行，僅供設計時看畫面) #Preview { ContentView() }
